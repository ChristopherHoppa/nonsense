# Specify the UNC path to the SharePoint drive
$uncPath = "\\server\share"

# Specify a specific file within the SharePoint drive to test
$fileToProcess = "$uncPath\example.txt"

# Check if the specified file exists
if (-not (Test-Path $fileToProcess)) {
    Write-Host "The specified file does not exist: $fileToProcess"
    exit
}

# Get the file object
$file = Get-Item -Path $fileToProcess

# Get the current date and time
$currentDateTime = Get-Date

# Attempt to update the last accessed time of the file
try {
    $file.LastAccessTime = $currentDateTime
    Write-Host "Last access time for $($file.FullName) has been updated successfully."
} catch {
    Write-Host "Error updating last access time for $($file.FullName): $_"
}
